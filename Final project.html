Creating sliders (with libraries in two colors)

In [21]:
for year in years:
    frame = {'data': [], 'name': str(year)}
    colors=[]
    difference_in_visitors=[]
    if(int(year)>1984):
        i=0
        for visitors in df[year]:
            year_before = str(int(year)-1)
            visitors_year_before=df[year_before][i]
            difference_in_visitors.append(df['library'][i]+"<br>Visitors this year: "+str(round(visitors/1000))+ "K<br>Difference from last year:"+ str(round((int(visitors)-int(visitors_year_before))/1000))+"K"),
            if(visitors_year_before>visitors):
                colors.append('#e27d60')
            else:
                colors.append('#41b3a3')
                
            i=i+1
    else:
        colors='#e27d60'
    library = go.Scattermapbox(
        lon = df['lon'],
        lat = df['lat'],
        text = difference_in_visitors,
        hoverinfo= 'text',
        marker=go.scattermapbox.Marker(
            size=df[year]/scale,
            color=colors
        ),
        showlegend=False,
    )
    frame['data'].append(library)
    figure['frames'].append(frame)
    slider_step = {'args': [
        [year],
        {'frame': {'duration': 1000, 'redraw': False},
         'mode': 'slow',
       'transition': {'duration': 1000}}
     ],
     'label': year,
     'method': 'animate'}
    sliders_dict['steps'].append(slider_step)
figure['layout']['sliders'] = [sliders_dict]
